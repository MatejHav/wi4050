\section{$\rho$-GNF Model}
% =============================================================================

% ── 5. Gaussian Copula as ρ ───────────────────────────────────────────────────
\begin{frame}{Gaussian Copula as Sensitivity Parameter}
  \textit{Model $\mathrm{corr}(z_A,z_Y)=\rho$ via a Gaussian copula. True $\rho$ is never identified.}
  \[
    C_\rho(u,v) \;=\; \Phi_\rho\!\bigl(\Phi^{-1}(u),\;\Phi^{-1}(v)\bigr)
  \]
  \[
    (z_A,\; z_Y) \;\sim\; \mathcal{N}\!\left(\mathbf{0},\;
    \Sigma_\rho = \begin{pmatrix}1 & \rho \\ \rho & 1\end{pmatrix}\right)
  \]
  \[
    \rho = 0: \text{ no confounding} \qquad |\rho| \to 1: \text{ maximal}
  \]
\end{frame}

% ── 6. ρ-GNF Generative Model ─────────────────────────────────────────────────
\begin{frame}{$\rho$-GNF: Generative Model}
  \textit{Correlated latents flow through a learned causal bijection $f_\theta$.}
  \begin{columns}[c]
    \column{0.46\textwidth}
    \begin{tikzpicture}[
      node distance=0.9cm and 1.5cm,
      every node/.style={circle,draw,thick,minimum size=0.75cm,font=\normalsize},
      >=Latex,thick
    ]
      \node (zA) {$z_A$};
      \node[below=0.9cm of zA] (zY) {$z_Y$};
      \node[right=1.6cm of zA] (A) {$A$};
      \node[right=1.6cm of zY] (Y) {$Y$};
      \draw[<->,blue,thick] (zA) -- (zY)
            node[midway,left,draw=none,blue,font=\small] {$\rho$};
      \draw[->] (zA) -- (A)
            node[midway,above,draw=none,font=\tiny] {$f_A^{-1}$};
      \draw[->] (zY) -- (Y)
            node[midway,below,draw=none,font=\tiny] {$f_Y^{-1}$};
      \draw[->] (A) -- (Y)
            node[midway,right,draw=none,font=\small] {$\tau$};
    \end{tikzpicture}
    \column{0.54\textwidth}
    \[
      p(A,Y) \;=\; p_Z\!\bigl(f(A,Y)\bigr)\,\bigl|\det J_f\bigr|
    \]
    \smallskip
    $f_\theta$ learned end-to-end via max.\ likelihood
  \end{columns}
\end{frame}

% ── 7. ATE Estimator ─────────────────────────────────────────────────────────
\begin{frame}{ATE Estimator via Flow Inversion}
  \textit{Set $A$ in data space; draw $z_Y$ from the correlated prior.}
  \[
    \hat\tau
    \;=\; \frac{1}{S}\sum_{s=1}^{S}
    \Bigl[
      \bigl[f^{-1}(\mathbf{z}^{(s)},\; A{=}1)\bigr]_Y
      \;-\;
      \bigl[f^{-1}(\mathbf{z}^{(s)},\; A{=}0)\bigr]_Y
    \Bigr]
  \]
  \[
    (z_A^{(s)},\; z_Y^{(s)}) \;\sim\; \mathcal{N}(\mathbf{0},\;\Sigma_\rho)
  \]
\end{frame}

\section{Experiments}
% =============================================================================

% ── 8. Setting 0: DGP ────────────────────────────────────────────────────────
\begin{frame}{Setting 0: Gaussian DGP (Sanity Check)}
  \textit{Model is correctly specified. Does it recover $\rho^*$?}
  \[
    \begin{pmatrix}e_A \\ e_Y\end{pmatrix}
    \sim \mathcal{N}\!\left(\mathbf{0},\;
    \begin{pmatrix}1 & \rho^* \\ \rho^* & 1\end{pmatrix}\right)
    ,\quad A = e_A,\quad Y = 0.2\, A + e_Y
  \]
  \[
    \rho^* = 0.30,\qquad \text{True ATE} = 0.20
  \]
\end{frame}

% ── 9. Setting 0: Results ─────────────────────────────────────────────────────
\begin{frame}{Setting 0: Results}
  \textit{Minimum RMSE at assumed $\rho \approx \rho^*$. Model is identifiable.}
  \begin{center}
    \includegraphics[width=0.92\textwidth]{../figures/setting0_rho_curve.png}
  \end{center}
\end{frame}

% ── 10. Setting 1: Clayton DGP ────────────────────────────────────────────────
\begin{frame}{Setting 1: Clayton Copula (Misspecification)}
  \textit{True copula has lower-tail dependence; Gaussian model is wrong.}
  \[
    C_\theta(u,v) \;=\; \bigl(u^{-\theta}+v^{-\theta}-1\bigr)^{-1/\theta},
    \qquad \theta = 2,\quad \lambda_L = 2^{-1/\theta}
  \]
  \[
    \rho_{\mathrm{ref}}
      = \sin\!\left(\frac{\pi}{2}\cdot\frac{\theta}{\theta+2}\right)
      \approx 0.707,
    \qquad \text{True ATE} = 0.20
  \]
\end{frame}

% ── 11. Setting 1: Results ────────────────────────────────────────────────────
\begin{frame}{Setting 1: Results}
  \textit{Copula mismatch shifts optimal $\rho$ and inflates RMSE.}
  \begin{center}
    \includegraphics[width=0.92\textwidth]{../figures/setting1_rho_curve.png}
  \end{center}
\end{frame}

% ── 12. Why Did Setting 1 Fail? ──────────────────────────────────────────────
\begin{frame}{Why Did Setting 1 Fail?}
  \textit{Gaussian shape forced for every DGP.}
  \[
    (z_A,z_Y)\sim\mathcal{N}(\mathbf{0},\Sigma_\rho)
    \quad\text{for \emph{every} dataset}
  \]
  \begin{itemize}
    \item No tail dependence: Clayton has $\lambda_L > 0$, Gaussian has $\lambda_L = 0$
    \item Optimal $\rho$ shifts away from $\rho_{\mathrm{ref}}$ under misspecification
  \end{itemize}
\end{frame}

% ── 13. A Harder Problem: Heterogeneous Confounding ──────────────────────────
\begin{frame}{A Harder Problem: Heterogeneous Confounding}
  \[
    (z_{A,i},\,z_{Y,i})\;\sim\;\mathcal{N}\!\left(\mathbf{0},\;\Sigma_{\textcolor{red}{\rho}}\right)
    \quad \forall\; i
  \]
  \begin{center}\itshape
    Same copula, same $\rho$, for every observation.
  \end{center}
  \bigskip
  \pause
  \begin{center}\itshape
    With many features, homogeneous confounding is almost guaranteed to break.
  \end{center}
\end{frame}

\section{Extension}

% ── 13b. Motivation: Per-Category Setup ──────────────────────────────────────
\begin{frame}{Motivation: One Model Per Category}
  \textit{For discrete $X$: run the setup separately for each group.}
  \begin{columns}[c]
    \column{0.5\textwidth}
    \centering\textbf{$X = \text{low}$}\\[10pt]
    \[
      (z_A,z_Y)\sim\mathcal{N}\!\left(\mathbf{0},\Sigma_{\rho_{\text{low}}}\right)
    \]
    \centering weak confounding

    \column{0.5\textwidth}
    \centering\textbf{$X = \text{high}$}\\[10pt]
    \[
      (z_A,z_Y)\sim\mathcal{N}\!\left(\mathbf{0},\Sigma_{\rho_{\text{high}}}\right)
    \]
    \centering strong confounding
  \end{columns}
  \medskip
  \begin{center}\itshape
    V2 generalises this: feed $X$ as context to a single flow.
  \end{center}
\end{frame}

% ── 14. Extension: Context Conditioning ──────────────────────────────────────
\begin{frame}{Extension: Context Conditioning}
  \textit{\textbf{V1}: no context \quad \textbf{V2}: condition on $X$}
  \begin{center}
    \includegraphics[width=0.95\textwidth]{../figures/setting2_comparison.png}
  \end{center}
  V1 and V2 nearly identical: context conditioning does not fix the copula.
\end{frame}

% ── 15. Further Extension: ρ(X) Conditional Copula ──────────────────────────
\begin{frame}{Further Extension: Covariate-Dependent Copula}
  \textit{V2 fixes the copula; V3 learns $\rho(X)$.}
  \begin{columns}[c]
    \column{0.46\textwidth}
    \centering\textbf{V2}\\[8pt]
    $\rho$ global; $X$ in flow only.
    \[
      z\sim\mathcal{N}\!\left(\mathbf{0},\,\Sigma_{\textcolor{red}{\rho}}\right)
    \]

    \column{0.54\textwidth}
    \centering\textbf{V3}\\[8pt]
    \[
      \rho(X) = \tanh\!\bigl(\mathrm{MLP}(X)\bigr)
    \]
    \[
      z\sim\mathcal{N}\!\left(\mathbf{0},\,\Sigma_{\textcolor{blue}{\rho(X)}}\right)
    \]
  \end{columns}
  \medskip
  \begin{center}\itshape
    Practitioner specifies a function, not a scalar. Models heterogeneous confounding.
  \end{center}
\end{frame}

% ── 16. Conclusion ───────────────────────────────────────────────────────────
\begin{frame}{Conclusion}
  \setbeamercovered{transparent}
  \begin{itemize}[<+->]
    \item Use $\rho$-GNF to bound ATE sensitivity to confounding.
    \item Works when the Gaussian copula assumption holds.
    \item Assumptions are strong; require expert knowledge to argue.
    \item Not robust to copula misspecification (Setting~1).
    \item Extended the flow to condition on $X$ (V2).
    \item V2 relaxes the homogeneous-confounding assumption.
    \item \alert{But shifts more uncheckable work to the practitioner.}
  \end{itemize}
\end{frame}

\begin{frame}{Backup: DAGConditioner (Causal Order Encoding)}
  \textit{Mask $M$ ensures $h_i$ sees only causal parents.}
  \[
    h_i \;=\; \mathrm{MLP}\!\bigl(M_{i,:} \odot x\bigr),
    \qquad M_{ij} = \mathbf{1}\bigl[j \in \mathrm{pa}(i)\bigr]
  \]
  For the graph $A \to Y$:
  \[
    h_A = \mathrm{MLP}(\mathbf{0}),
    \qquad
    h_Y = \mathrm{MLP}(A)
  \]
  Graph $A \to Y$ is fixed; only flow weights $\theta$ are trained.
\end{frame}

\begin{frame}{Backup: UMNN (Unconstrained Monotone Neural Network)}
  \textit{Each variable maps through a strictly monotone bijection.}
  \[
    z_i \;=\; \int_0^{x_i} g_\phi\!\bigl(t;\; h_i\bigr)\;\mathrm{d}t \;+\; b_i,
    \qquad g_\phi > 0
  \]
  \[
    \log\left|\frac{\partial z_i}{\partial x_i}\right|
    \;=\; \log g_\phi(x_i;\; h_i)
  \]
  $g_\phi$: MLP $+$ \texttt{softplus}. Jacobian is exact, not approximated.
\end{frame}

\begin{frame}{Backup: Training Objective}
  \textit{Maximize log-likelihood; $\Sigma_\rho$ fixed.}
  \[
    \mathcal{L}(\theta)
    \;=\; \frac{1}{N}\sum_{n=1}^N
    \Bigl[
      \log p_Z\!\bigl(z^{(n)}\bigr)
      \;+\;
      \log\bigl|\det J_f(x^{(n)})\bigr|
    \Bigr]
  \]
  \[
    p_Z(z) \;=\; \mathcal{N}(z;\;\mathbf{0},\;\Sigma_\rho)
  \]
  Only the flow parameters $\theta$ are optimized; $\rho$ is a hyperparameter.
\end{frame}
